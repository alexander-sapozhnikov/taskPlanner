CREATE TABLE users
(
    id SERIAL PRIMARY KEY,
    username VARCHAR(90) UNIQUE,
    password varchar(90),
    firstname VARCHAR(90),
    lastname VARCHAR(90),
    phone VARCHAR(20)
);

CREATE TABLE tasks
(
    id SERIAL PRIMARY KEY,
    name VARCHAR(90),
    description text,
    alert_time date,
    alert_received boolean
);

CREATE TABLE listsOfUsers
(
    id SERIAL PRIMARY KEY,
    userId INT8 REFERENCES users(id),
    name varchar(210)
);

CREATE TABLE tasksInLists
(
    id SERIAL PRIMARY KEY,
    listId INT8 REFERENCES listsOfUsers(id),
    taskId INT8 REFERENCES tasks(id),
    unique(listId, taskId)
);


CREATE TABLE watcherForTasks
(
    id SERIAL PRIMARY KEY,
    contactId INT8 REFERENCES tasks(id),
    userId INT8 REFERENCES users(id),
    unique(contactId, userId)
);
